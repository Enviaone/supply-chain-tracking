  _id: ObjectId,

  module: "production_stage_entries",

  recordId: 1234,

  action: "CREATE | UPDATE | DELETE",

  batchId: 45,
  stageId: 3,

  oldValue: {...},
  newValue: {...},

  comment: "admin correction",

  userId: 5,

  ipAddress: "192.168.1.10",

}

Indexes

{ recordId: 1 }
{ batchId: 1 }
{ timestamp: -1 }


alerts

Used for mismatch alerts.

{
  _id: ObjectId,

  type: "MISMATCH",

  batchId: 45,

  stageId: 2,

  expectedQty: 100,

  enteredQty: 120,

  recipients: ["+91xxxx"],

  whatsappSent: true,

  createdAt: ISODate()
}
